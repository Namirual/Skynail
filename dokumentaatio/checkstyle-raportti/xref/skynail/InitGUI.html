38      public static void main(String[] args) {
39  
40          JFrame frame = new JFrame("Skynail");
41          frame.setPreferredSize(new Dimension(450, 450));
42          frame.setSize(450, 450);
43  
44          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
45  
46          Road a = new Road("Test 1", new MapPoint(20, 50));
47          Road b = new Road("Test 2", new MapPoint(60, 110));
48          Road c = new Road("Test 3", new MapPoint(70, 160));
49          Road d = new Road("Test 4", new MapPoint(120, 110));
50          Road e = new Road("Test 5", new MapPoint(150, 130));
51          Road f = new Road("Test 6", new MapPoint(190, 110));
52          Road g = new Road("Test 7", new MapPoint(180, 150));
53  
54          a.addPointsBothWays(b);
55          b.addPointsBothWays(c, d);
56          c.addPointsBothWays(d);
57          d.addPointsBothWays(e);
58          e.addPointsBothWays(f, g);
59  
60          List<Point> worldPoints = new ArrayList<>();
61  
62          worldPoints.addAll(Arrays.asList(a, b, c, d, e, f, g));
63          
64          Team player = new Team("Pelaaja", a);
65          MapSwingPainter mapPainter = new MapSwingPainter(worldPoints, player);
66  
67          DiceRoller diceRoller = new RandomRoller();
68          
69          MapController controller = new MapController(player, diceRoller, mapPainter);
70          controller.handlePointInput(a);
71          mapPainter.setController(controller);
72          
73          frame.getContentPane().add(mapPainter);
74          frame.setVisible(true);
75      }
