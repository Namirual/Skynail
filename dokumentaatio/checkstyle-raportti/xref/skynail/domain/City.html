6   package skynail.domain;
7   
8   import java.util.ArrayList;
9   import java.util.HashMap;
10  import java.util.List;
11  import skynail.gui.MapPoint;
12  
13  /**
14   * Map point for cities, presently unfinished.
15   *
16   * @author lmantyla
17   */
18  public class City extends Road implements Point {
19  
20      String introText;
21      HashMap<Item, Integer> items;
22  
23      /**
24       * Creates new City.
25       *
26       * @param name name of the city.
27       */
28      public City(String name) {
29          super(name);
30      }
31  
32      /**
33       * Creates new City with introduction text.
34       *
35       * @param name Name of the city.
36       * @param introText Introduction text while entering the city.
37       */
38      public City(String name, String introText) {
39          super(name);
40          this.introText = introText;
41          this.items = new HashMap<Item, Integer>();
42      }
43  
44      /**
45       * Creates new City for the graphical user interface, MapPoint is included.
46       *
47       * @param name Name of the city.
48       * @param introText Introduction text while entering the dungeon.
49       * @param mapPoint mapPoint containing the coordinates of the city.
50       */
51      public City(String name, String introText, MapPoint mapPoint) {
52          super(name, mapPoint);
53          this.introText = introText;
54          this.items = new HashMap<Item, Integer>();
55      }
56  
57      /**
58       * Adds item to the list of items sold in the city.
59       * @param item Item being added.
60       * @param price Selling price of the item.
61       */
62      public void addItem(Item item, int price) {
63          items.put(item, price);
64      }
65  
66      public HashMap<Item, Integer> getItems() {
67          return items;
68      }
69  
70      public String getIntroText() {
71          return introText;
72      }
73  }
