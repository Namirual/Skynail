6   package skynail.game;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Map;
11  
12  import skynail.domain.Point;
13  import skynail.domain.Team;
14  import skynail.gui.MapPainter;
15  import skynail.service.PathService;
16  import skynail.service.DiceRoller;
17  
18  /**
19   *
20   * @author lmantyla
21   */
22  public class MapController {
23  
24      private Team player;
25      private DiceRoller diceRoller;
26      private MapPainter mapPainter;
27      private Map<Point, Integer> legalMoves;
28      private int moves;
29  
30      public MapController(Team player, DiceRoller diceRoller, MapPainter mapPainter) {
31          this.player = player;
32          this.diceRoller = diceRoller;
33          this.mapPainter = mapPainter;
34  
35          PathService pathService = new PathService(player);
36          this.legalMoves = pathService.calculateLegalMoves(1);
37      }
38  
39      // This function is called by MapListener, and supplies
40      // MapPainter with information about available legal moves.
41      public void handlePointInput(Point point) {
42          List<Point> legalMoveList = new ArrayList<Point>();
43          legalMoveList.addAll(legalMoves.keySet());
44  
45          if (legalMoves.containsKey(point)) {
46              player.setLocation(point);
47              moves = 0;
48          }
49  
50          update();
51      }
52  
53      public int handleDiceRoll() {
54          moves = diceRoller.diceThrow(3);
55          update();
56          return moves;
57      }
58  
59      public void update() {
60  
61          PathService pathService = new PathService(player);
62          legalMoves = pathService.calculateLegalMoves(moves);
63              
64          List<Point> legalMoveList = new ArrayList<Point>();
65          legalMoveList.addAll(legalMoves.keySet());
66  
67          mapPainter.setLegalMoves(legalMoveList);
68          mapPainter.update();
69  
70      }
71  }
