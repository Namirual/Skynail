6   package skynail.game;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Map;
11  
12  import skynail.domain.City;
13  import skynail.domain.Dungeon;
14  
15  import skynail.domain.Point;
16  import skynail.domain.Player;
17  import skynail.gui.UIManager;
18  import skynail.service.PathService;
19  import skynail.service.DiceRoller;

21  /**
22   * Controller with methods that receives and handles player input from the UI.
23   *
24   * @author lmantyla
25   */
26  public class MapController {
27  
28      private Player player;
29      private DiceRoller diceRoller;
30  
31      private UIManager uiManager;
32  
33      PathService pathService;
34  
35      List<Point> worldMap;
36      List<Point> pathPoints;
37  
38      private Map<Point, Integer> legalMoves;
39      private int moves;
40      boolean moving;
41  
42      /**
43       * Initializes the Map Controller.
44       *
45       * @param player
46       * @param uiManager
47       * @param diceRoller
48       */
49      public MapController(Player player, List<Point> worldMap, DiceRoller diceRoller, UIManager uiManager) {
50          this.player = player;
51          this.diceRoller = diceRoller;
52          this.uiManager = uiManager;
53          this.worldMap = worldMap;
54  
55          this.pathService = new PathService(player);
56          this.legalMoves = pathService.calculateLegalMoves(1);
57  
58          this.pathPoints = new ArrayList<Point>();
59          this.moving = false;
60      }
61  
62      /**
63       * The method attempts to move the player to point selected in the UI.
64       * <p>
65       * The method is called by the user interface. The method subsequently calls
66       * update() to initiate redrawing the map view.
67       *
68       * @param point Map point selected by player in the interface.
69       */
70      public void handlePointInput(Point point) {
71  
72          if (legalMoves.containsKey(point)) {
73  
74              moving = true;
75              pathPoints = pathService.getMovementPath(point);
76              uiManager.displayMapMovement(pathPoints);
77  
78              player.setLocation(point);
79              moving = false;
80  
81              moves = 0;
82              legalMoves.clear();
83  
84              if (point.getClass().equals(City.class)) {
85                  uiManager.startCityScene((City) point);
86              }
87  
88              if (point.getClass().equals(Dungeon.class)) {
89                  Dungeon dungeon = (Dungeon) point;
90                  BattleController battleController = new BattleController(uiManager, player, dungeon.getMonster());
91                  BattleState bState = battleController.startBattle();
92                  if (bState == BattleState.victory) {
93                      player.setGold(player.getGold() + 150);
94                  }
95              }
96          }
97      }
98  
99      /**
100      * Calls a remote dice roller.
101      *
102      * @return random number from dice roller
103      */
104     public int handleDiceRoll() {
105         moves = diceRoller.diceThrow(3);
106         legalMoves = pathService.calculateLegalMoves(moves);
107 
108         return moves;
109     }
110 
111     /**
112      * Used to make purchases in cities, currently unfinished.
113      */
114     public boolean handleCityBuy() {
115         return player.buyWithGold(100);
116     }
117 
118     public Player getPlayer() {
119         return player;
120     }
121 
122     public void setPlayer(Player player) {
123         this.player = player;
124     }
125 
126     public List<Point> getWorldMap() {
127         return worldMap;
128     }
129 
130     public Map<Point, Integer> getLegalMoves() {
131         return legalMoves;
132     }
133 
134     public List<Point> getPathPoints() {
135         return pathPoints;
136     }
137 
138     public boolean isMoving() {
139         return moving;
140     }
141 
142     public void setWorldMap(List<Point> worldMap) {
143         this.worldMap = worldMap;
144     }
145 }
