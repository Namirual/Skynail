42      @Override
43      public void mouseClicked(MouseEvent e) {
44          tieto.setText(e.getX() + " " + e.getY());
45  
46          for (Point point : worldMap) {
47              if (point.getMapPoint().checkIfInside(e.getX(), e.getY())) {
48                  tieto.setText("Point Clicked! " + e.getX() + " " + e.getY());
49                  manager.getMapController().handlePointInput(point);
50  
51                  return;
52                  
53              }
54          }
55      }
