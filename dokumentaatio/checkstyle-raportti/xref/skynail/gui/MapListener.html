40      @Override
41      public void mouseClicked(MouseEvent e) {
42          tieto.setText(e.getX() + " " + e.getY());
43  
44          for (Point point : manager.getMapController().getWorldMap()) {
45              if (point.getMapPoint().checkIfInside(e.getX(), e.getY())) {
46                  tieto.setText("Point Clicked! " + e.getX() + " " + e.getY());
47                  manager.getMapController().handlePointInput(point);
48                  manager.getMapPainter().update();
49                  return;
50                  
51              }
52          }
53      }
