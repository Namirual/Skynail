6   package skynail.gui;
7   
8   import java.awt.BorderLayout;
9   import java.awt.CardLayout;
10  import java.util.List;
11  import javax.swing.JPanel;
12  import skynail.domain.City;
13  import skynail.domain.Point;
14  import skynail.game.MapController;
15  
16  /**
17   * Graphical user interface used to handle changes between different views.
18   *
19   * @author lmantyla
20   */
21  public class GUIManager extends JPanel implements UIManager {
22  
23      MapPainter mapPainter;
24      MapListener mapListener;
25      MapController mapController;
26      CityScene scene;
27  
28      public GUIManager(List<Point> worldMap) {
29          setSize(600, 600);
30          setLayout(new BorderLayout());
31  
32          MapPainter mapPainter = new MapPainter(worldMap, this);
33          this.setMapPainter(mapPainter);
34      }
35  
36      public MapPainter getMapPainter() {
37          return mapPainter;
38      }
39  
40      public void setMapPainter(MapPainter mapPainter) {
41          this.mapPainter = mapPainter;
42          this.add(mapPainter);
43      }
44  
45      public void updateMap(List<Point> legalMoveList) {
46          mapPainter.setLegalMoves(legalMoveList);
47          mapPainter.update();
48  
49      }
50  
51      public void startCityScene(City city) {
52          scene = new CityScene(this, city);
53          this.add(scene);
54          mapPainter.setVisible(false);
55          scene.setVisible(true);
56      }
57  
58      public void endCityScene(City city) {
59          mapPainter.setVisible(true);
60          scene.setVisible(false);
61      }
62  
63      public MapController getMapController() {
64          return mapController;
65      }
66  
67      public void setMapController(MapController mapController) {
68          this.mapController = mapController;
69      }
70  }
