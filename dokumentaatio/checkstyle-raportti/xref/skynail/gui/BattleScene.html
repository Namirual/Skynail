6   package skynail.gui;
7   
8   import java.awt.Dimension;
9   import java.awt.GridBagLayout;
10  import java.awt.GridLayout;
11  import java.awt.event.ActionEvent;
12  import java.awt.event.ActionListener;
13  import javax.swing.BoxLayout;
14  import javax.swing.JButton;
15  import javax.swing.JLabel;
16  import javax.swing.JPanel;
17  import javax.swing.JTable;
18  import skynail.domain.City;
19  import skynail.domain.Companion;
20  import skynail.game.BattleController;
21  import skynail.game.BattleState;
22  
23  /**
24   *
25   * @author lmantyla
26   */
27  public class BattleScene extends JPanel {
28  
29      private GUIManager manager;
30  
31      private JLabel label;
32      private JLabel label2;
33      private JButton button1;
34      private JButton button2;
35      private BattleController controller;
36  
37      private JPanel companionPanel;
38      private JPanel monsterPanel;
39      private JPanel buttonPanel;
40  
41      BattleScene(GUIManager manager, BattleController controller) {
42  
43          this.controller = controller;
44  
45          label = new JLabel("Monster HP: " + controller.getMonster().getHP());
46          label2 = new JLabel("Fight it out!");
47  
48          monsterPanel = new JPanel();
49          monsterPanel.setLayout(new GridLayout(1, 0));
50  
51          monsterPanel.add(label);
52          companionPanel = new JPanel();
53  
54          JPanel messagePanel = new JPanel();
55          messagePanel.add(label2);
56  
57          JPanel statPanel = new JPanel();
58          statPanel.setLayout(new GridLayout(1, 2));
59          statPanel.setPreferredSize(new Dimension(450, 50));
60          statPanel.add(monsterPanel);
61          statPanel.add(companionPanel);
62  
63          writeCompanions();
64  
65          this.add(statPanel);
66          this.add(messagePanel);
67  
68          button1 = new JButton("Attack!");
69          button2 = new JButton("Escape!");
70  
71          button1.addActionListener(new ActionListener() {
72              @Override
73              public void actionPerformed(ActionEvent e) {
74                  BattleState state = controller.handlePlayerAttack(controller.getMonster());
75                  if (state == BattleState.playerTurn) {
76                      writeCompanions();
77                      label.setText("Monster HP " + controller.getMonster().getHP());
78                      label2.setText("You hit the monster!");
79                  }
80  
81                  if (state == BattleState.enemyTurn) {
82                      writeCompanions();
83                      label.setText("Monster HP " + controller.getMonster().getHP());
84                      label2.setText("You hit the monster, and the monster strikes back!");
85                  }
86  
87                  if (state == BattleState.victory) {
88                      label.setText("Monster HP " + controller.getMonster().getHP());
89                      victoryScene();
90                  }
91                  if (state == BattleState.death) {
92                      deathScene();
93                  }
94              }
95          });
96  
97          button2.addActionListener(new ActionListener() {
98              @Override
99              public void actionPerformed(ActionEvent e) {
100                 manager.endBattleScene();
101             }
102         });
103 
104         buttonPanel = new JPanel();
105         buttonPanel.add(button1);
106         buttonPanel.add(button2);
107         this.add(buttonPanel);
108     }
109 
110     public void writeCompanions() {
111         companionPanel.removeAll();
112         companionPanel.setLayout(new GridLayout(controller.getPlayer().getCompanions().size(), 1));
113 
114         for (int luku = 0; luku < controller.getPlayer().getCompanions().size(); luku++) {
115             Companion companion = controller.getPlayer().getCompanions().get(luku);
116             JPanel charPanel = new JPanel();
117             JLabel charLabel;
118             if (controller.getCharacterTurn() == luku) {
119                 charLabel = new JLabel("<html><b>" + "<br>" + companion.getName() + " HP: " + companion.getHP() + "</b>");
120             } else {
121                 charLabel = new JLabel("<html>" + "<br>" + companion.getName() + " HP: " + companion.getHP());
122             }
123 
124             companionPanel.add(charLabel);
125         }
126     }
127 
128     public void victoryScene() {
129         label2.setText("You won!");
130         buttonPanel.remove(button1);
131         button2.setText("Leave victorious!");
132     }
133 
134     public void deathScene() {
135         label.setText("You died...");
136         buttonPanel.remove(button1);
137         buttonPanel.remove(button2);
138     }
139 }
