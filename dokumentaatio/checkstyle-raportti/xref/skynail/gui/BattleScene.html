116     public void checkState(BattleState state) {
117         if (state == BattleState.playerTurn) {
118             label2.setText("You hit the monster!");
119         }
120 
121         if (state == BattleState.enemyTurn) {
122             label2.setText("You hit the monster, and monsters strikes back!");
123         }
124 
125         if (state == BattleState.victory) {
126             victoryScene();
127         }
128         if (state == BattleState.death) {
129             deathScene();
130         }
131     }

133     public void update() {
134         graphicPanel.removeAll();
135         
136         monsterPanel.removeAll();
137 
138         monsterPanel.setLayout(new GridLayout(controller.getMonsters().size(), 1));
139         int monsterHeight = 50;
140 
141         for (int num = 0; num < controller.getMonsters().size(); num++) {
142             Monster monster = controller.getMonsters().get(num);
143             final int finalNum = num;
144             JButton character = new JButton(monster.getName());
145             character.setBounds(50, 50, 100, 50);
146 
147             character.addActionListener(new ActionListener() {
148                 @Override
149                 public void actionPerformed(ActionEvent e) {
150                     BattleState state = controller.handlePlayerAttack(controller.getMonsters().get(finalNum));
151                     checkState(state);
152                     update();
153                 }
154             });
155 
156             character.setLocation(100, monsterHeight);
157             monsterHeight += 50;
158             graphicPanel.add(character);
159 
160             JLabel monsterLabel = new JLabel("<html><b>" + "<br>" + monster.getName() + " HP: " + monster.getHP() + "</b>");
161             monsterPanel.add(monsterLabel);
162         }
163 
164         companionPanel.removeAll();
165         companionPanel.setLayout(new GridLayout(controller.getPlayer().getCompanions().size(), 1));
166 
167         int charHeight = 50;
168 
169         for (int num = 0; num < controller.getPlayer().getCompanions().size(); num++) {
170             Companion companion = controller.getPlayer().getCompanions().get(num);
171 
172             int finalNum = num;
173 
174             JButton character = new JButton(companion.getName());
175             character.setBounds(50, 50, 100, 50);
176 
177             character.setLocation(300, charHeight);
178             charHeight += 50;
179             graphicPanel.add(character);
180 
181             character.addActionListener(new ActionListener() {
182                 @Override
183                 public void actionPerformed(ActionEvent e) {
184                     if (currentItem == null) {
185                         return;
186                     }
187                     BattleState state = controller.handleItemUse(controller.getPlayer().getCompanions().get(finalNum), currentItem);
188                     currentItem = null;
189                     label1.setText("No item selected.");
190                     update();
191                     if (state == BattleState.playerTurn) {
192                         label2.setText("You heal " + controller.getPlayer().getCompanions().get(finalNum).getName());
193                     }
194                 }
195             });
196 
197             JPanel charPanel = new JPanel();
198             JLabel charLabel;
199             if (controller.getCharacterTurn() == num) {
200                 charLabel = new JLabel("<html><b>" + "<br>" + companion.getName() + " HP: " + companion.getHP() + "</b>");
201             } else {
202                 charLabel = new JLabel("<html>" + "<br>" + companion.getName() + " HP: " + companion.getHP());
203             }
204 
205             companionPanel.add(charLabel);
206 
207             companionPanel.repaint();
208             companionPanel.revalidate();
209 
210             monsterPanel.repaint();
211             monsterPanel.revalidate();
212 
213             graphicPanel.repaint();
214             graphicPanel.revalidate();
215         }
216     }

218     public void itemSelection() {
219         JDialog itemSelect = new JDialog();
220 
221         itemSelect.setPreferredSize(new Dimension(240, 240));
222         itemSelect.setSize(240, 240);
223         itemSelect.setLocationRelativeTo(this);
224         itemSelect.setModal(true);
225         itemSelect.setUndecorated(true);
226 
227         JButton leaveButton = new JButton("Return");
228         JPanel itemPanel = new JPanel();
229         itemSelect.getContentPane().add(itemPanel);
230         itemPanel.add(leaveButton);
231 
232         leaveButton.addActionListener(new ActionListener() {
233             @Override
234             public void actionPerformed(ActionEvent e) {
235                 itemSelect.setVisible(false);
236             }
237         });
238 
239         for (Item item : controller.getPlayer().getItems().keySet()) {
240             int itemNumber = controller.getPlayer().getItems().get(item);
241 
242             JButton itemButton = new JButton(item.getName() + " : " + itemNumber);
243             
244             itemPanel.add(itemButton);
245             itemButton.addActionListener(new ActionListener() {
246                 @Override
247                 public void actionPerformed(ActionEvent e) {
248                     currentItem = item;
249                     label1.setText("Current item: " + item.getName() + " : " + itemNumber);
250                     itemSelect.setVisible(false);
251                 }
252             });
253         }
254         itemSelect.setVisible(true);
255     }

257     public void victoryScene() {
258 
259         JDialog victoryMessage = new JDialog();
260 
261         victoryMessage.setPreferredSize(new Dimension(240, 240));
262         victoryMessage.setSize(240, 240);
263         victoryMessage.setLocationRelativeTo(this);
264         victoryMessage.setModal(true);
265         victoryMessage.setUndecorated(true);
266 
267         JPanel victoryTexts = new JPanel();
268         JButton endButton = new JButton("Leave victorious!");
269         JLabel endText = new JLabel();
270 
271         endButton.addActionListener(new ActionListener() {
272             @Override
273             public void actionPerformed(ActionEvent e) {
274                 victoryMessage.setVisible(false);
275                 manager.getMapController().processBattleResult(BattleState.victory);
276                 manager.endBattleScene();
277                 victoryMessage.dispose();
278             }
279         });
280         buttonPanel.remove(button1);
281 
282         endText.setText("You won the battle!");
283 
284         victoryTexts.add(endText);
285         victoryTexts.add(endButton);
286         victoryMessage.getContentPane().add(victoryTexts);
287 
288         victoryMessage.setVisible(true);
289     }
290 
291     public void deathScene() {
292         label2.setText("You died...");
293         buttonPanel.remove(button1);
294         buttonPanel.remove(button2);
295     }
