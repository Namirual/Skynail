27  public class MapPainter extends JPanel {
28  
29      MapListener listener;
30      GUIManager manager;
31  
32      public MapPainter(GUIManager manager) {
33          this.manager = manager;
34  
35          JPanel panel = new JPanel();
36          this.add(panel);
37          setSize(450, 450);
38  
39          JTextField field = new JTextField("Pixel coordinates of clicks.");
40          panel.add(field);
41  
42          listener = new MapListener(field, manager);
43          this.addMouseListener(listener);
44  
45          JButton button = new JButton("Roll dice.");
46  
47          button.addActionListener(new ActionListener() {
48              @Override
49              public void actionPerformed(ActionEvent e) {
50                  int roll = manager.getMapController().handleDiceRoll();
51                  field.setText("Your dice roll is " + roll);
52                  update();
53              }
54          });
55  
56          //button.setBounds(20, 20, 100, 30);
57          panel.add(button);
58      }
59  
60      public void movePath(List<Point> characterPath) {
61          for (int luku = characterPath.size() - 1; luku > 0; luku--) {
62              moveInALine(characterPath.get(luku), characterPath.get(luku - 1),
63                      manager.getMapController().getPlayer().getMapPoint());
64          }
65      }
66  
67      public void moveInALine(Point origin, Point target, MapPoint characterPosition) {
68          for (int luku = 0; luku < 20; luku++) {
69              characterPosition.setXf(characterPosition.getXf()
70                      + (target.getMapPoint().getX() - origin.getMapPoint().getX()) / 20f);
71              characterPosition.setYf(characterPosition.getYf()
72                      + (target.getMapPoint().getY() - origin.getMapPoint().getY()) / 20f);
73              try {
74                  Thread.sleep(20);
75              } catch (InterruptedException ex) {
76                  Thread.currentThread().interrupt();
77              }
78              this.paintImmediately(0, 0, 450, 450);
79          }
80  
81          characterPosition = new MapPoint(target.getMapPoint().getX(), target.getMapPoint().getY());
82      }
83  
84      public void paint(Graphics g) {
85          super.paint(g);
86  
87          List<Point> legalMoves = new ArrayList<Point>();
88          legalMoves.addAll(manager.getMapController().getLegalMoves().keySet());
89  
90          for (Point point : manager.getMapController().getWorldMap()) {
91              Graphics2D g2 = (Graphics2D) g;
92              Graphics2D g3 = (Graphics2D) g;
93  
94              for (Point targetPoint : point.getLinkedPoints()) {
95                  Line2D line = new Line2D.Float(point.getMapPoint().getX(), point.getMapPoint().getY(),
96                          targetPoint.getMapPoint().getX(), targetPoint.getMapPoint().getY());
97                  g3.setColor(Color.black);
98                  g3.draw(line);
99              }
100             if (legalMoves.contains(point)) {
101                 g2.setColor(Color.BLUE);
102             } else {
103                 g2.setColor(Color.gray);
104             }
105             g2.fillOval((point.getMapPoint().getX() - 10), point.getMapPoint().getY() - 10, 20, 20);
106         }
107         Graphics2D g4 = (Graphics2D) g;
108 
109         MapPoint characterPosition = manager.getMapController().getPlayer().getMapPoint();
110 
111         if (characterPosition == null) {
112             Point point = manager.getMapController().getPlayer().getLocation();
113             g4.setColor(Color.red);
114             g4.fillRect((point.getMapPoint().getX() - 10), point.getMapPoint().getY() - 10, 20, 20);
115         } else {
116             g4.setColor(Color.red);
117             g4.fillRect((characterPosition.getX() - 10), characterPosition.getY() - 10, 20, 20);
118         }
119 
120     }
121 
122     public void update() {
123         repaint();
124     }
125 
126     public MapListener getMapListener() {
127         return listener;
128     }
