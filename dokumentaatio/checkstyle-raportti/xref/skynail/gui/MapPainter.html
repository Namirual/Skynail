24  /**
25   * Extended JPanel used to initialise and draw the map screen.
26   *
27   * @author lmantyla
28   */
29  public class MapPainter extends JPanel {
30  
31      MapListener listener;
32      GUIManager manager;
33  
34      JPanel buttonPanel;
35      JButton rollDice;
36      JButton explore;
37  
38      final int sizex = 480;
39      final int sizey = 480;
40  
41      public MapPainter(GUIManager manager) {
42          this.manager = manager;
43          setSize(sizex, sizey);
44  
45          buttonPanel = new JPanel();
46          this.add(buttonPanel);
47  
48          JTextField field = new JTextField("Pixel coordinates of clicks.");
49          buttonPanel.add(field);
50  
51          listener = new MapListener(field, manager);
52          this.addMouseListener(listener);
53  
54          rollDice = new JButton("Roll dice");
55          explore = new JButton("Enter area");
56  
57          rollDice.addActionListener(new ActionListener() {
58              @Override
59              public void actionPerformed(ActionEvent e) {
60                  int roll = manager.getMapController().handleDiceRoll();
61                  field.setText("Your dice roll is " + roll);
62                  update();
63              }
64          });
65  
66          explore.addActionListener(new ActionListener() {
67              @Override
68              public void actionPerformed(ActionEvent e) {
69                  manager.getMapController().handleEnteringArea();
70              }
71          });
72  
73          buttonPanel.add(rollDice);
74          buttonPanel.add(explore);
75          explore.setVisible(false);
76  
77      }
78  
79      public void movePath(List<Point> characterPath) {
80          buttonPanel.setVisible(false);
81          for (int luku = characterPath.size() - 1; luku > 0; luku--) {
82              moveInALine(characterPath.get(luku), characterPath.get(luku - 1),
83                      manager.getMapController().getPlayer().getMapPoint());
84          }
85          updateGUI(characterPath.get(0));
86  
87          buttonPanel.setVisible(true);
88      }
89  
90      public void moveInALine(Point origin, Point target, MapPoint characterPosition) {
91          for (int luku = 0; luku < 20; luku++) {
92              characterPosition.setXf(characterPosition.getXf()
93                      + (target.getMapPoint().getX() - origin.getMapPoint().getX()) / 20f);
94              characterPosition.setYf(characterPosition.getYf()
95                      + (target.getMapPoint().getY() - origin.getMapPoint().getY()) / 20f);
96              try {
97                  Thread.sleep(20);
98              } catch (InterruptedException ex) {
99                  Thread.currentThread().interrupt();
100             }
101             this.paintImmediately(0, 0, sizex, sizey);
102         }
103 
104         characterPosition = new MapPoint(target.getMapPoint().getX(), target.getMapPoint().getY());
105     }
106 
107     public void updateGUI(Point currentPoint) {
108         if (currentPoint.getClass().equals(Dungeon.class)) {
109             explore.setText("Enter dungeon");
110             explore.setVisible(true);
111         } else if (currentPoint.getClass().equals(City.class)) {
112             explore.setText("Enter city");
113             explore.setVisible(true);
114         } else {
115             explore.setVisible(false);
116         }
117     }
118 
119     public void paint(Graphics g) {
120         super.paint(g);
121 
122         List<Point> legalMoves = new ArrayList<Point>();
123         legalMoves.addAll(manager.getMapController().getLegalMoves().keySet());
124 
125         for (Point point : manager.getMapController().getWorldMap()) {
126             Graphics2D g2 = (Graphics2D) g;
127             Graphics2D g3 = (Graphics2D) g;
128 
129             for (Point targetPoint : point.getLinkedPoints()) {
130                 Line2D line = new Line2D.Float(point.getMapPoint().getX(), point.getMapPoint().getY(),
131                         targetPoint.getMapPoint().getX(), targetPoint.getMapPoint().getY());
132                 g3.setColor(Color.black);
133                 g3.draw(line);
134             }
135             if (legalMoves.contains(point)) {
136                 g2.setColor(Color.BLUE);
137             } else {
138                 g2.setColor(Color.gray);
139             }
140             if (point.getClass().equals(Road.class)) {
141                 g2.fillOval((point.getMapPoint().getX() - 10), point.getMapPoint().getY() - 10, 20, 20);
142             } else {
143                 g2.fillOval((point.getMapPoint().getX() - 15), point.getMapPoint().getY() - 15, 30, 30);
144                 g2.setColor(Color.WHITE);
145                 g2.fillOval((point.getMapPoint().getX() - 10), point.getMapPoint().getY() - 10, 20, 20);
146             }
147         }
148         Graphics2D g4 = (Graphics2D) g;
149 
150         MapPoint characterPosition = manager.getMapController().getPlayer().getMapPoint();
151 
152         if (characterPosition == null) {
153             Point point = manager.getMapController().getPlayer().getLocation();
154             g4.setColor(Color.red);
155             g4.fillRect((point.getMapPoint().getX() - 10), point.getMapPoint().getY() - 10, 20, 20);
156         } else {
157             g4.setColor(Color.red);
158             g4.fillRect((characterPosition.getX() - 10), characterPosition.getY() - 10, 20, 20);
159         }
160 
161     }
162 
163     public void update() {
164         repaint();
165     }
166 
167     public MapListener getMapListener() {
168         return listener;
169     }
