23  /**
24   * Generates the city view and buttons.
25   *
26   * @author lmantyla
27   */
28  public class CityScene extends JPanel {
29  
30      CityScene(UIManager manager, City city) {
31  
32          //setSize(450, 450);
33          JLabel label = new JLabel(city.getIntroText());
34          JLabel label2 = new JLabel("");
35          this.add(label);
36          this.add(label2);
37  
38          JButton button1 = new JButton("Buy things");
39          JButton button2 = new JButton("Leave");
40  
41          button1.addActionListener(new ActionListener() {
42              @Override
43              public void actionPerformed(ActionEvent e) {
44                  if (manager.getMapController().handleCityBuy()) {
45                      label2.setText("You bought an item!");
46                  } else {
47                      label2.setText("You don't have enough gold!");
48                  }
49              }
50          });
51  
52          button2.addActionListener(new ActionListener() {
53              @Override
54              public void actionPerformed(ActionEvent e) {
55                  manager.endCityScene(city);
56              }
57          });
58  
59          this.add(button1);
60          this.add(button2);
61      }
62  }
